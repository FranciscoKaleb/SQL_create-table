import RPi.GPIO as GPIO
import tkinter as tk

# --- GPIO Setup ---
LED_PIN = 17
GPIO.setmode(GPIO.BCM)
GPIO.setup(LED_PIN, GPIO.OUT)
GPIO.output(LED_PIN, GPIO.LOW)

# --- GUI Setup ---
window = tk.Tk()
window.title("LED Control")

def turn_on():
    GPIO.output(LED_PIN, GPIO.HIGH)
    status_label.config(text="LED is ON", fg="green")

def turn_off():
    GPIO.output(LED_PIN, GPIO.LOW)
    status_label.config(text="LED is OFF", fg="red")

on_button = tk.Button(window, text="Turn ON", command=turn_on, width=15, height=2)
on_button.pack(pady=10)

off_button = tk.Button(window, text="Turn OFF", command=turn_off, width=15, height=2)
off_button.pack(pady=10)

status_label = tk.Label(window, text="LED is OFF", fg="red", font=("Arial", 12))
status_label.pack(pady=10)

window.mainloop()

# Cleanup GPIO on close
GPIO.cleanup()
